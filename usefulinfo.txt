Then:

if event.type == pygame.KEYDOWN:
    if event.key == pygame.K_UP:

!!!
Different shapes take different arguments to draw

pygame.draw.circle(screen, color, [x, y], radius)
pygame.draw.rect(screen, color,[x ,y ,wdith, height])
!!!


!!!
distance_to() is used to check distance between two things. For example:
 if i.pos.distance_to(Player.pos) < 20:
    mobs.remove(i)

good here because since the values are floats they might never equal if i do:
if i.pos == Player.pos:
!!!


!!!
"font = pygame.font.Font(x.ttf, 50)" returns a font object

It has the methods such as .render(text, antialiasing, color, backgroundcolor)
!!!


!!!
screen.blit means copy one image into another

screen.blit(source, position)

can be used to display text or images.

player_img = pygame.image.load("player.png")
screen.blit(player_img, (300, 400))

health_text = font.render(f"Health {Player.health}", True, (255, 255, 255))
screen.blit(health_text, (20, 20))
